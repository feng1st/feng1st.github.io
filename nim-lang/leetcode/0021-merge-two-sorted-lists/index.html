[Nim4LeetCode] 21. Merge Two Sorted Lists

Posted on 2018, Mar 27 One min read

Problem

Merge two sorted linked lists and return it as a new list. The new list should be made by splicing together the nodes of the first two lists.

Example:

Input: 1->2->4, 1->3->4
Output: 1->1->2->3->4->4

Solution

type
  ListNodeObj = object
    val: int
    next: ref ListNodeObj
  ListNode = ref ListNodeObj

proc newListNode(val: int, next: ListNode = nil): ListNode =
  result = ListNode(val: val, next: next)

proc mergeTwoLists(l1: ListNode, l2: ListNode): ListNode =
  let dummy = newListNode(0)
  var
    p = dummy
    p1 = l1
    p2 = l2
  while p1 != nil and p2 != nil:
    if p1.val < p2.val:
      p.next = p1
      p1 = p1.next
    else:
      p.next = p2
      p2 = p2.next
    p = p.next
  p.next = if p1 == nil: p2
    else: p1
  result = dummy.next

when isMainModule:
  let
    l1 = newListNode(1, newListNode(2, newListNode(4)))
    l2 = newListNode(1, newListNode(3, newListNode(4)))
    l = mergeTwoLists(l1, l2)
  assert l.val == 1 and
    l.next.val == 1 and
    l.next.next.val == 2 and
    l.next.next.next.val == 3 and
    l.next.next.next.next.val == 4 and
    l.next.next.next.next.next.val == 4 and
    l.next.next.next.next.next.next == nil
