[Nim4LeetCode] 7. Reverse Integer

Posted on 2018, Mar 13 One min read

Problem

Given a 32-bit signed integer, reverse digits of an integer.

Example 1:

Input: 123
Output:  321

Example 2:

Input: -123
Output: -321

Example 3:

Input: 120
Output: 21

Note:
Assume we are dealing with an environment which could only hold integers within the 32-bit signed integer range. For the purpose of this problem, assume that your function returns 0 when the reversed integer overflows.

Solution

const
  INT_MIN = 0x80000000'i32
  INT_MAX = 0x7FFFFFFF'i32
  INT_MIN_DIV_10 = INT_MIN div 10
  INT_MAX_DIV_10 = INT_MAX div 10

proc reverse(x: int32): int32 =
  var
    t = x
    y = 0'i32
  while t != 0:
    if x >= 0:
      if y > INT_MAX_DIV_10 or
        y * 10 > INT_MAX - t mod 10:
        return 0
    else:
      if y < INT_MIN_DIV_10 or
        y * 10 < INT_MIN - t mod 10:
        return 0
    y = y * 10 + t mod 10
    t = t div 10
  result = y

when isMainModule:
  assert reverse(123) == 321
  assert reverse(-123) == -321
  assert reverse(120) == 21
  assert reverse(1463847412) == 2147483641
  assert reverse(-1463847412) == -2147483641
  assert reverse(1563847412) == 0
  assert reverse(-1563847412) == 0
