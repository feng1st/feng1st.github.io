[Nim4LeetCode] 22. Generate Parentheses

Posted on 2018, Mar 28 One min read

Problem

Given n pairs of parentheses, write a function to generate all combinations of well-formed parentheses.

For example, given n = 3, a solution set is:

[
  "((()))",
  "(()())",
  "(())()",
  "()(())",
  "()()()"
]

Solution

proc backtrack(res: var seq[string], str: string, n, open, close: int) =
  if open == n and close == n:
    res.add(str)
    return
  if open < n:
    backtrack(res, str & '(', n, open + 1, close)
  if close < open:
    backtrack(res, str & ')', n, open, close + 1)

proc generateParenthesis(n: int): seq[string] =
  result = newSeq[string]()
  backtrack(result, "", n, 0, 0)

when isMainModule:
  assert generateParenthesis(3) ==
    @["((()))", "(()())", "(())()", "()(())", "()()()"]
