[Nim4LeetCode] 5. Longest Palindromic Substring

Posted on 2018, Mar 11 One min read

Problem

Given a string s, find the longest palindromic substring in s. You may assume that the maximum length of s is 1000.

Example:

Input: "babad"

Output: "bab"

Note: "aba" is also a valid answer.

Example:

Input: "cbbd"

Output: "bb"

Solution

proc expandAroundCenter(s: string, left: int, right: int): (int, int) =
  var
    l = left
    r = right
  while l >= 0 and r < s.len and s[l] == s[r]:
    l -= 1
    r += 1
  result = (l + 1, r - 1)

proc longestPalindrome(s: string): string =
  var
    left = 0
    right = 0
  for i in 0 ..< s.len:
    let (l, r) = expandAroundCenter(s, i, i)
    if r - l > right - left:
      (left, right) = (l, r)
  for i in 0 .. s.len - 2:
    let (l, r) = expandAroundCenter(s, i, i + 1)
    if r - l > right - left:
      (left, right) = (l, r)
  result = s[left .. right]

when isMainModule:
  assert longestPalindrome("babad") == "bab" or
    longestPalindrome("babad") == "aba"
  assert longestPalindrome("cbbd") == "bb"
